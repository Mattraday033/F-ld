using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class MapPathCanvasSizeAdjuster : MonoBehaviour
{
	public Canvas mapPathCanvas;
	
	public RectTransform backgroundCanvasRectTransform;
	public RectTransform backgroundUIRectTransform;
	
	public RectTransform mapPathCanvasRectTransform;

	private void Awake()
	{
		mapPathCanvas.renderMode = RenderMode.WorldSpace;
	}

    // Update is called once per frame
    void Update() //here for Animation
    {
        mapPathCanvasRectTransform.position = backgroundCanvasRectTransform.position;
		mapPathCanvasRectTransform.localScale = backgroundCanvasRectTransform.localScale;
		
		mapPathCanvasRectTransform.sizeDelta = backgroundUIRectTransform.sizeDelta;
    }
}
